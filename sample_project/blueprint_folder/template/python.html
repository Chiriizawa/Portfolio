    def count_vowels(vow):
        vowels = 'aeiou'
        count = 0
        for a in vow:
            if a in vowels:
                count += 1
        return count
    
    assert count_vowels("banana") == 3
    assert count_vowels("ubuntu") == 3
    assert count_vowels("mango") == 2
    assert count_vowels("QQQQ") == 0
    assert count_vowels("wyrd") == 0
    print("Exercise 45 is correct.")

    def grade():
        try:
            prelim = float(input("Enter Your Prelim Grade: "))
            midterm = float(input("Enter Your Midterm Grade: "))
            finals = float(input("Enter Your Finals Grade: "))
        except:
            print("Invalid")
            return
        else:
            gwa = (prelim + midterm + finals) / 3
            if gwa > 100:
                print("Invalid")
                return
            if gwa >= 75:
                print("Passed")
            else:
                print("Failed")
        finally:
            print(f"GWA", gwa)
    grade()

    def solution(prelim, midterm, finals):
        grade  (prelim * 0.30) + (midterm * 0.30) + (finals * 0.40)
        if grade != int(grade):
            grade = int(grade) + 1
        return round(grade)
